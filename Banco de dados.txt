    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cargo VARCHAR(50) NOT NULL,
    departamento VARCHAR(50),
    email VARCHAR(100) UNIQUE,
    telefone VARCHAR(20) UNIQUE
);
